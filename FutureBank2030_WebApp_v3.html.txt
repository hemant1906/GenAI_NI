function makeSparkline(id, vals, color) {
  const el = document.getElementById(id);
  if (!el) return;
  const max = Math.max(...vals), min = Math.min(...vals);
  vals.forEach((v, i) => {
    const bar = document.createElement('div');
    bar.className = 'spark-bar';
    const h = Math.max(3, ((v - min) / (max - min + 0.01)) * 28);
    bar.style.height = h + 'px';
    bar.style.background = i === vals.length - 1 ? color : color + '88';
    el.appendChild(bar);
  });
}

  });

// ── Response Router ──
function getScriptedResponse(text) {
  const l = text.toLowerCase();

  // 1. Check follow-up specific overrides
  for (const [key, val] of Object.entries(followUpDB)) {
    if (l.includes(key)) return val;
  }

  // 2. Check blocked triggers (red chip queries)
  for (const [key, val] of Object.entries(blockedTriggers)) {
    if (l.includes(key)) return val;
  }

  // 3. Standard keyword routing to aiDB
  if (l.includes('spend')||l.includes('analys')||l.includes('subscri')||l.includes('coffee')) return aiDB.spending;
  if (l.includes('sav')&&(l.includes('rate')||l.includes('interest')||l.includes('account'))) return aiDB.savings;
  if (l.includes('nvidia')||l.includes('stock')||(l.includes('invest')&&!l.includes('goal'))||l.includes('etf')||l.includes('trade')) return aiDB.nvidia;
  if (l.includes('fraud')||l.includes('alert')||l.includes('login')||l.includes('security')||l.includes('block')) return aiDB.fraud;
  if (l.includes('house')||l.includes('home')||l.includes('mortgage')||l.includes('deposit')) return aiDB.house;
  if (l.includes('tax')||l.includes('deduct')||l.includes('ato')||l.includes('wfh')||l.includes('return')) return aiDB.tax;
  if (l.includes('quit')||l.includes('career')||l.includes('part-time')||l.includes('part time')||l.includes('job')||l.includes('simulat')||l.includes('leave')) return aiDB.quit;
  if (l.includes('goal')||l.includes('james')||l.includes('italy')||l.includes('joint')||l.includes('partner')) return aiDB.goals;
  if (l.includes('crypto')||l.includes('bitcoin')||l.includes('ethereum')||l.includes('btc')||l.includes('eth')) return aiDB.crypto;
  if (l.includes('smsf')||l.includes('self managed')) {
    return { verdict:'BLOCKED', rule:'ASIC RG 175 — SMSF advice requires specific AFSL authorisation.', blockedText:'Nova can\'t provide personal SMSF advice.', why:'SMSF advice is Tier 1 personal financial product advice under ASIC RG 175, requiring AFSL with SMSF-specific authorisation which FutureBank\'s AI Advisor does not hold.', canDo:'Nova can share general SMSF educational information. For personal advice, contact a licensed SMSF financial adviser.' };
  }
  return aiDB.default;
}

function renderResponse(data, queryText) {
  logCompliance(queryText, data.verdict, data.rule);
  if (data.verdict === 'BLOCKED') {
    addBlockedBubble(data.blockedText, data.why, data.canDo);
  } else if (data.verdict === 'MODIFIED') {
    addModifiedBubble(data.modifiedResponse, data.complianceNote);
    addFollowUpChips(data.followUps);
  } else {
    addApprovedBubble(data.novaResponse, data.rule);
    addFollowUpChips(data.followUps);
  }
}

function sendChat() {
  const inp = document.getElementById('chatInput');
  const txt = inp.value.trim();
  if (!txt) return;
  inp.value = '';
  addUserBubble(txt);
  const data = getScriptedResponse(txt);
  const phase1 = 900 + Math.random() * 600;
  const phase2 = 650 + Math.random() * 450;
  showTypingPhase(1);
  setTimeout(() => {
    showTypingPhase(2);
    setTimeout(() => {
      removeTyping();
      renderResponse(data, txt);
    }, phase2);
  }, phase1);
}

function fmtK(n){
  if(Math.abs(n)>=1000000)return'A$'+(n/1000000).toFixed(1)+'M';
  if(Math.abs(n)>=1000)return'A$'+Math.round(n/1000)+'K';
